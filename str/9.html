          Запись.Часов = 0;
          Запись.Дней  = 0;
        Иначе	          
          Запись.Часов = 8;
          Запись.Дней  = 1;
        КонецЕсли; 
        Дат = Дат + ЧислоСекундВСутках;
      КонецЦикла; 
      Набор.Записать();
